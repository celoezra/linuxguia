SAMBA INSTALAÇÃO E CONFIGURAÇÃO

apt-get install samba

configuração do arquivo samba > /etc/samba/smb.conf

[global]

workgroup = cosmos

server string = neturno

server role = standalone server

passdb backend = smbpasswd

security = user

map to guest = bad user

[arquivos]

path = /home/share

guest ok = yes

read only = no

comment = MY Files

public = yes

create mask = 0700

directory mask = 0700

rowseable = yes

writable = yes

printable = no

create mode = 0777

directory mode = 0777
